                <div class="list-item nav-item focused" data-type="group" onclick="loadChannels('live')">Canlı TV</div>
                <div class="list-item nav-item" data-type="group" onclick="loadChannels('vod')">Filmler (VOD)</div>
                <div class="list-item nav-item" data-type="group" onclick="loadChannels('series')">Diziler</div>

                    </div>

    <script>
        // ES5 Kısıtlamalarına Uygun Değişkenler
        var currentFocusIndex = 0;
        var navItems = [];
       
        // Sahte API verisi (Sen bunu XMLHttpRequest ile değiştireceksin)
        var dummyData = {
            live: ['TRT 1', 'ATV', 'Kanal D', 'Show TV', 'Star TV', 'TV8', 'NTV', 'Habertürk'],
            vod: ['Matrix', 'Yüzüklerin Efendisi', 'Inception', 'Interstellar', 'Gladyatör'],
            series: ['Breaking Bad', 'Game of Thrones', 'Ezel', 'Kurtlar Vadisi']
        };

        // Tuş Kodları (webOS)
        var KEYS = { LEFT: 37, UP: 38, RIGHT: 39, DOWN: 40, ENTER: 13 };

        window.onload = function() {
            document.addEventListener('keydown', handleKeyDown);
            loadChannels('live'); // Varsayılan olarak Canlı TV'yi yükle
            updateNavItems();
            updateClock();
            setInterval(updateClock, 60000);
        };

        function updateClock() {
            var date = new Date();
            var hours = date.getHours();
            var minutes = date.getMinutes();
            if (hours < 10) hours = '0' + hours;
            if (minutes < 10) minutes = '0' + minutes;
            document.getElementById('clock').innerHTML = hours + ':' + minutes;
        }

        // --- İÇERİK YÜKLEME ---
        function loadChannels(category) {
            var grid = document.getElementById('channel-grid');
            var title = document.getElementById('content-title');
            grid.innerHTML = ''; // Temizle
           
            if (category === 'live') title.innerHTML = 'Canlı TV Kategorileri';
            if (category === 'vod') title.innerHTML = 'Filmler (VOD)';
            if (category === 'series') title.innerHTML = 'Diziler';

            var items = dummyData[category] || [];

            for (var i = 0; i < items.length; i++) {
                var itemDiv = document.createElement('div');
                itemDiv.className = 'grid-item nav-item';
                itemDiv.setAttribute('data-type', 'channel');
                itemDiv.innerHTML = '<span>' + items[i] + '</span>';
                grid.appendChild(itemDiv);
            }

            // Yeni kanallar yüklendiğinde navItems'i güncelle ama odağı sol menüde tut
            updateNavItems(true);
        }

        // --- SPATIAL NAVIGATION (TV KUMANDA ODAĞI) ---
        function updateNavItems(keepFocusIndex) {
            var elements = document.getElementsByClassName('nav-item');
            navItems = [];
            for (var i = 0; i < elements.length; i++) {
                navItems.push(elements[i]);
            }
            if (!keepFocusIndex) currentFocusIndex = 0;
            applyFocus();
        }

        function applyFocus() {
            if (navItems.length === 0) return;
           
            for (var i = 0; i < navItems.length; i++) {
                navItems[i].className = navItems[i].className.replace(' focused', '');
            }
           
            if (currentFocusIndex < 0) currentFocusIndex = 0;
            if (currentFocusIndex >= navItems.length) currentFocusIndex = navItems.length - 1;
           
            navItems[currentFocusIndex].className += ' focused';
        }

        function handleKeyDown(e) {
            var keyCode = e.keyCode || e.which;
            var currentElement = navItems[currentFocusIndex];
            var isSidebar = currentElement.getAttribute('data-type') === 'group';

            if (keyCode === KEYS.DOWN) {
                if (isSidebar) {
                    // Sidebar'da aşağı in
                    if (currentFocusIndex < 3) currentFocusIndex++;
                } else {
                    // Grid'de aşağı in (ortalama 3 kolon)
                    currentFocusIndex += 3;
                }
                applyFocus();
                e.preventDefault();
            }
            else if (keyCode === KEYS.UP) {
                if (isSidebar) {
                    if (currentFocusIndex > 0) currentFocusIndex--;
                } else {
                    currentFocusIndex -= 3;
                }
                applyFocus();
                e.preventDefault();
            }
            else if (keyCode === KEYS.RIGHT) {
                if (isSidebar) {
                    // Sağ menüye (Grid'e) atla
                    currentFocusIndex = 4; // Sidebar'da 4 eleman var (0,1,2,3), 4. indeks ilk kanaldır
                } else {
                    currentFocusIndex++;
                }
                applyFocus();
                e.preventDefault();
            }
            else if (keyCode === KEYS.LEFT) {
                if (!isSidebar) {
                    if (currentFocusIndex % 3 === 1) { // Eğer grid'in en solundaysa (1, 4, 7 indexleri)
                        // Sidebar'a geri dön (aktif olan gruba)
                        currentFocusIndex = 0; // Şimdilik basite indirgedik, 0'a döner
                    } else {
                        currentFocusIndex--;
                    }
                }
                applyFocus();
                e.preventDefault();
            }
            else if (keyCode === KEYS.ENTER) {
                if (currentElement) currentElement.click();
                e.preventDefault();
            }
        }
